Defined in src/ajax.js:188

Available since 0.1.0

Send an AJAX GET Request

Parameters:

  • strOpt String | Object

    File path or Options

  • callback Function | Object

    The function to execute

  • Options Object

Example:

$.get({
  url: "path/to/file.js",
  success: function (data) {
    console.log(data);
  }
}); // Longer form, the below is preferred

$.get("path/to/file.js", function (data) {
  console.log(data);
}); // This does the exact same function as above

$.get("path/to/file.js", function (data) {
  console.log(data);
}, {
  error: function (err) {
    console.error(err);
  }
}); // Shortform, with more options
